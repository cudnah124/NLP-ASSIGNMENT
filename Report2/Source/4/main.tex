\section{Tác vụ 2.3: Phân loại Intent Mệnh đề (Clause Intent Classification)}

\subsection{Đặt vấn đề}

Ngoài việc biết \textit{ai làm gì}, hệ thống cần hiểu \textbf{chức năng pháp lý} của từng mệnh đề trong hợp đồng. Một mệnh đề có thể:
\begin{itemize}
    \item Áp đặt nghĩa vụ bắt buộc (\textbf{Obligation})
    \item Cấm một hành động (\textbf{Prohibition})
    \item Trao quyền cho một bên (\textbf{Right})
    \item Xác định điều kiện chấm dứt hợp đồng (\textbf{Termination Condition})
\end{itemize}

\subsection{Dataset}

\subsection{Mô hình Baseline: TF-IDF + Logistic Regression}

\subsubsection{Cấu hình}

\begin{itemize}
    \item \textbf{TF-IDF Vectorizer}: \texttt{max\_features=5000}, \texttt{ngram\_range=(1,2)}, \texttt{stop\_words="english"}
    \item \textbf{Logistic Regression}: \texttt{C=1.0}, \texttt{class\_weight="balanced"}, \texttt{max\_iter=1000}
    \item Đánh giá bằng 5-fold cross-validation trên tập train.
\end{itemize}

\subsubsection{Nhận xét lỗi phổ biến}

\begin{itemize}
    \item \textbf{Prohibition vs. Obligation}: Một số câu dạng \textit{``shall not''} bị phân loại nhầm thành Obligation vì từ khoá \textit{shall} chiếm trọng số lớn trong TF-IDF.
    \item \textbf{Right vs. Obligation}: Cả hai đều có thể bắt đầu bằng \textit{``shall be entitled to''}, dẫn đến nhầm lẫn khi không đủ context.
\end{itemize}

\subsection{Nhận xét tổng thể}

\begin{itemize}
    \item Mô hình TF-IDF + LogReg đạt accuracy 91\% -- hiệu quả cao so với độ phức tạp thấp và không cần GPU.
    \item DistilBERT có khả năng nắm bắt ngữ cảnh tốt hơn nhờ attention mechanism, kỳ vọng cải thiện trên các nhãn khó (Prohibition, Right).
    \item Cả hai mô hình đều lợi dụng được từ khoá pháp lý đặc thù: \textit{shall, may, must not, entitled to, terminate}.
\end{itemize}
